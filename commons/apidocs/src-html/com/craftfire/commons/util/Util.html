035        /**
036         * Checks if the string is an email.
037         *
038         * @param string The email.
039         * @return true, if the string is an email.
040         */
041        public static boolean isEmail(String string) {
042            if (string != null && !string.isEmpty()) {
043                Pattern p = Pattern.compile(".+@.+\\.[a-z]+");
044                Matcher m = p.matcher(string);
045                if (m.matches()) {
046                    return true;
047                }
048            }
049            return false;
050        }
051    
052        /**
053         * Checks if string is a valid IP. Wildcards (*) are allowed.
054         *
055         * @param string The IP as a string.
056         * @return true, if string is an IP.
057         */
058        public static boolean isIP(String string) {
059            if (string != null && !string.isEmpty()) {
060                String[] parts = string.split("\\.");
061                if (parts.length == 4) {
062                    for (String s : parts) {
063                        if (s.equals("*")) {
064                            continue;
065                        }
066                        if (!isInteger(s)) {
067                            return false;
068                        }
069                        int i = Integer.parseInt(s);
070                        if (i < 0 || i > 255) {
071                            return false;
072                        }
073                    }
074                    return true;
075                }
076            }
077            return false;
078        }

080        /**
081         * Checks if input is an integer.
082         *
083         * @param input The string to parse/check.
084         * @return true, if the input is an integer.
085         */
086        public static boolean isInteger(String input) {
087            try {
088                Integer.parseInt(input);
089                return true;
090            } catch (Exception e) {
091                return false;
092            }
093        }

096         * Checks if input is a Long object.
097         *
098         * @param input The string to parse/check.
099         * @return true, if the input is a Long object.
100         */
101        public static boolean isLong(String input) {
102            try {
103                Long.parseLong(input);
104                return true;
105            } catch (Exception e) {
106                return false;
107            }
108        }

110        public static boolean isURLOnline(URL url) {
111            return Util.getResponseCode(url) == 200;
112        }
113    
114        public static boolean isURLOnline(String urlString) {
115            URL url = isValidURL(urlString);
116            return url != null && isURLOnline(url);
117        }
118    
119        public static URL isValidURL(String urlString) {
120            try {
121                return new URL(urlString);
122            } catch (MalformedURLException e) {
123                return null;
124            }
125        }
126    
127        public static int getResponseCode(String urlString) {
128            try {
129                return getResponseCode(new URL(urlString));
130            } catch (MalformedURLException e) {
131                return 0;
132            }
133        }
134    
135        public static int getResponseCode(URL url) {
136            try {
137                HttpURLConnection connection = (HttpURLConnection) url.openConnection();
138                connection.setConnectTimeout(3000);
139                connection.setRequestMethod("GET");
140                connection.setRequestProperty("User-Agent", "Mozilla/5.0 (Windows; U; Windows NT 6.0; en-US; rv:1.9.1.2) Gecko/20090729 Firefox/3.5.2 (.NET CLR 3.5.30729)");
141                connection.connect();
142                return connection.getResponseCode();
143            } catch (IOException e) {
144                return 0;
145            }
146        }

148        public static String join(Object[] elements, String separator) {
149            if (elements == null || elements.length <= 0) {
150                return "";
151            }
152            StringBuilder builder = new StringBuilder();
153            for (int i = 0; i < elements.length; ++i) {
154                if (i > 0) {
155                    builder.append(separator);
156                }
157                if (elements[i] != null) {
158                    builder.append(elements[i]);
159                }
160            }
161            return builder.toString();
162        }
163    
164        public static String join(Collection<?> elements, String separator) {
165            if (elements == null || elements.isEmpty()) {
166                return "";
167            }
168            StringBuilder builder = new StringBuilder();
169            Iterator<?> itr = elements.iterator();
170            while (itr.hasNext()) {
171                builder.append(itr.next());
172                if (itr.hasNext()) {
173                    builder.append(separator);
174                }
175            }
176            return builder.toString();
177        }
