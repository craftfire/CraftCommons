020    package com.craftfire.commons.yaml;
021    
022    import java.io.File;
023    import java.util.List;
024    import java.util.Map;
025    import java.util.Set;
026    
027    import com.craftfire.commons.util.LoggingManager;
028    
029    /**
030     * An interface for all kinds of yaml managers.
031     */
032    public interface YamlManager {
033    
034        /**
035         * Returns a set of files used by the manager.
036         * 
037         * @return set of files
038         */
039        Set<File> getFiles();
040    
041        /**
042         * Returns the logging manager used by this manager.
043         * 
044         * @return the logging manager
045         */
046        LoggingManager getLogger();
047    
048        /**
049         * Sets the logging manager to be used by this manager.
050         * 
051         * @param  loggingManager            the logging manager
052         * @throws IllegalArguementException if the loggingManager is {@code null}
053         */
054        void setLoggingManager(LoggingManager loggingManager);
055    
056        /**
057         * Returns the root YamlNode of the document loaded by this manager.
058         * 
059         * @return the root YamlNode
060         */
061        YamlNode getRootNode();
062    
063        /**
064         * Replaces the root YamlNode of the document loaded by this manager with given node.
065         * 
066         * @param node  the node to replace the root node
067         * @return      the new root node
068         */
069        YamlNode setRootNode(YamlNode node);
070    
071        /**
072         * Checks if node with given path exists in the manager.
073         * 
074         * @param node  the node path to check
075         * @return      {@code true} if exists, {@code false} otherwise
076         */
077        boolean exist(String node);
078    
079        /**
080         * Returns {@code boolean} value of node with given path.
081         * <p>
082         * Defaults to {@code false} if the node doesn't exist or can't be converted to {@code boolean}.
083         * 
084         * @param node  the node path
085         * @return      {@code boolean} value of the node or {@code false} if not found or can't be converted
086         */
087        boolean getBoolean(String node);
088    
089        /**
090         * Returns {@code boolean} value of node with given path.
091         * 
092         * @param node          the node path
093         * @param defaultValue  the default value to use if the node doesn't exist or can't be converted to {@code boolean}
094         * @return              {@code boolean} value of the node or {@code defaultValue} if not found or can't be converted
095         */
096        boolean getBoolean(String node, boolean defaultValue);
097    
098        /**
099         * Returns {@code String} value of node with given path.
100         * <p>
101         * Defaults to {@code null} if the node doesn't exist or can't be converted to {@code String}.
102         * 
103         * @param node  the node path
104         * @return      {@code String} value of the node or {@code null} if not found or can't be converted
105         */
106        String getString(String node);
107    
108        /**
109         * Returns {@code String} value of node with given path.
110         * 
111         * @param node          the node path
112         * @param defaultValue  the default value to use if the node doesn't exist or can't be converted to {@code String}
113         * @return              {@code String} value of the node or {@code defaultValue} if not found or can't be converted
114         */
115        String getString(String node, String defaultValue);
116    
117        /**
118         * Returns {@code int} value of node with given path.
119         * <p>
120         * Defaults to {@code 0} if the node doesn't exist or can't be converted to {@code int}.
121         * 
122         * @param node  the node path
123         * @return      {@code int} value of the node or {@code null} if not found or can't be converted
124         */
125        int getInt(String node);
126    
127        /**
128         * Returns {@code int} value of node with given path.
129         * 
130         * @param node          the node path
131         * @param defaultValue  the default value to use if the node doesn't exist or can't be converted to {@code int}
132         * @return              {@code int} value of the node or {@code defaultValue} if not found or can't be converted
133         */
134        int getInt(String node, int defaultValue);
135    
136        /**
137         * Returns {@code long} value of node with given path.
138         * <p>
139         * Defaults to {@code 0} if the node doesn't exist or can't be converted to {@code long}.
140         * 
141         * @param node  the node path
142         * @return      {@code long} value of the node or {@code null} if not found or can't be converted
143         */
144        long getLong(String node);
145    
146        /**
147         * Returns {@code long} value of node with given path.
148         * 
149         * @param node          the node path
150         * @param defaultValue  the default value to use if the node doesn't exist or can't be converted to {@code long}
151         * @return              {@code long} value of the node or {@code defaultValue} if not found or can't be converted
152         */
153        long getLong(String node, long defaultValue);
154    
155        /**
156         * Adds all nodes from given YamlManager to this manager's document.
157         * 
158         * @param  yamlManager   the YamlManager to take nodes from
159         * @throws YamlException if the root node of the document is a scalar
160         */
161        void addNodes(YamlManager yamlManager) throws YamlException;
162    
163        /**
164         * Adds all nodes from given map to this manager's document.
165         * 
166         * @param  map           map of nodes to add
167         * @throws YamlException if the root node of the document is a scalar
168         */
169        void addNodes(Map<String, Object> map) throws YamlException;
170    
171        /**
172         * Sets node with the given path to given value.
173         * 
174         * @param  node          the node path
175         * @param  value         the value to set
176         * @throws YamlException if one of path elements is a scalar node
177         */
178        void setNode(String node, Object value) throws YamlException;
179    
180        /**
181         * Returns YamlNode with the given path.
182         * 
183         * @param  node          the node path
184         * @return               the YamlNode
185         * @throws YamlException if one of path elements is a scalar node
186         */
187        YamlNode getNode(String node) throws YamlException;
188    
189        /**
190         * Returns a map of the values of all the final nodes. The map key is the node path, the map value is the node value. 
191         * 
192         * @return a map of values of all the final nodes
193         */
194        @Deprecated
195        public Map<String, Object> getNodes();
196    
197        /**
198         * Returns list of all scalar nodes loaded by the manager.
199         * 
200         * @return list of all scalar nodes
201         */
202        List<YamlNode> getFinalNodeList();
203    
204        /**
205         * Returns total number of all scalar nodes loaded by the manager.
206         * 
207         * @return number of all scalar nodes
208         */
209        int getFinalNodeCount();
210    
211        /**
212         * Saves the manager's document to the file it was loaded from.
213         * <p>
214         * Fails if the document wasn't loaded from a file or IOException occurred.
215         * 
216         * @return {@code true} if succeed, {@code false} otherwise
217         */
218        boolean save();
219    
220        /**
221         * Loads the manager's document.
222         * <p>
223         * Fails if trying to load the document from a stream or a reader more than once, or IOException occurred.
224         * 
225         * @return {@code true} if succeed, {@code false} otherwise
226         */
227        boolean load();
228    
229    }
